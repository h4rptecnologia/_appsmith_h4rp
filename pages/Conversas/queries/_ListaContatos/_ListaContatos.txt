SELECT 
  -- Conversas
  c.id AS conversation_id,
  c.account_id AS conversation_account_id,
  c.inbox_id AS conversation_inbox_id,
  c.status AS conversation_status,
  c.assignee_id AS conversation_assignee_id,
  c.created_at AS conversation_created_at,
  c.updated_at AS conversation_updated_at,
  c.contact_id AS conversation_contact_id,
  c.display_id AS conversation_display_id,
  c.team_id AS conversation_team_id,

  -- Mensagens
  m.id AS message_id,
  m."content" AS message_content,
  m.created_at AS message_created_at,
  COALESCE(m.sender_type, 'Não Identificado') AS message_sender_type,

  -- Contatos
  ct.id AS contact_id,
  COALESCE(ct."name", 'Usuário não identificado') AS contact_name,
  COALESCE(ct.email, 'E-mail não identificado') AS contact_email,

  -- Inbox Members
  im.id AS inbox_member_id,
  im.user_id AS inbox_member_user_id,
  im.inbox_id AS inbox_member_inbox_id,
  im.created_at AS inbox_member_created_at,

  -- Teams
  t.id AS team_id,
  t."name" AS team_name,
  t.description AS team_description,
  t.allow_auto_assign AS team_allow_auto_assign,
  t.account_id AS team_account_id

FROM chatwoot.conversations c
INNER JOIN chatwoot.messages m 
  ON c.id = m.conversation_id
INNER JOIN chatwoot.contacts ct 
  ON ct.id = c.contact_id
LEFT JOIN chatwoot.inbox_members im 
  ON im.inbox_id = c.inbox_id
LEFT JOIN chatwoot.teams t 
  ON t.id = im.id;
