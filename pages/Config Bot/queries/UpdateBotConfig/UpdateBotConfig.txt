UPDATE public.bot_config
SET
    chatwoot_team_id = {{Number(SelectPromptTeam.selectedOptionValue)}},
    principal_assignee_id = 1,
    bot_name = {{InputBotName.text}},
    description = NULLIF({{InputDescription.text || ''}}, ''),
    enabled = {{SwitchEnabled.isSwitchedOn}},
    full_prompt = {{InputFullPrompt.text}},

    temperature = {{
InputTemperature.text && String(InputTemperature.text).trim() !== ''
? Number(InputTemperature.text)
: 0.30
}},

    allow_handoff = {{SwitchAllowHandoff.isSwitchedOn}},

    fallback_team_id = {{
SelectFallbackTeam.selectedOptionValue && String(SelectFallbackTeam.selectedOptionValue).trim() !== ''
? Number(SelectFallbackTeam.selectedOptionValue)
: null
}},

    updated_at = NOW()
WHERE id = {{appsmith.store.botConfigSelected.id || Table1.selectedRow.id}}
RETURNING *;
