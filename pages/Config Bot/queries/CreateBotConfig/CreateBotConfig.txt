INSERT INTO public.bot_config (
	chatwoot_team_id,
	principal_assignee_id,
	bot_name,
	description,
	enabled,
	full_prompt,
	temperature,
	allow_handoff,
	fallback_team_id,
	created_at,
	updated_at
)
VALUES (
	{{Number(SelectPromptTeam.selectedOptionValue)}},
	1,
	{{InputBotName.text}},
	NULLIF({{InputDescription.text || ''}}, ''),
	{{SwitchEnabled.isSwitchedOn}},
	{{InputFullPrompt.text}},
	COALESCE(NULLIF({{InputTemperature.text || ''}}, '')::NUMERIC, 0.30),
	{{SwitchAllowHandoff.isSwitchedOn}},
	{{SelectFallbackTeam.selectedOptionValue ? Number(SelectFallbackTeam.selectedOptionValue) : null}},
	NOW(),
	NOW()
)
ON CONFLICT (chatwoot_team_id) WHERE chatwoot_team_id IS NOT NULL
DO UPDATE SET
    principal_assignee_id = 1,
    bot_name = EXCLUDED.bot_name,
    description = EXCLUDED.description,
    enabled = EXCLUDED.enabled,
    full_prompt = EXCLUDED.full_prompt,
    provider = EXCLUDED.provider,
    model_name = EXCLUDED.model_name,
    temperature = EXCLUDED.temperature,
    max_output_tokens = EXCLUDED.max_output_tokens,
    allow_handoff = EXCLUDED.allow_handoff,
    fallback_team_id = EXCLUDED.fallback_team_id,
    updated_at = NOW()
RETURNING *;
