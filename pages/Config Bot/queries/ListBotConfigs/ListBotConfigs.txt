SELECT
    bc.id,
    bc.chatwoot_team_id,
    COALESCE(t_prompt.name, 'Time #' || bc.chatwoot_team_id::TEXT) AS chatwoot_team_name,
    COALESCE(bc.principal_assignee_id, 1) AS principal_assignee_id,
    bc.bot_name,
    bc.description,
    bc.enabled,
    CASE
        WHEN bc.full_prompt IS NULL THEN ''
        WHEN LENGTH(bc.full_prompt) > 220 THEN LEFT(bc.full_prompt, 220) || '...'
        ELSE bc.full_prompt
    END AS prompt_resumo,
    bc.full_prompt,
    bc.provider,
    bc.model_name,
    bc.temperature,
    bc.max_output_tokens,
    bc.allow_handoff,
    bc.fallback_team_id,
    t_fallback.name AS fallback_team_name,
    bc.created_at,
    bc.updated_at
FROM public.bot_config bc
LEFT JOIN chatwoot.teams t_prompt
    ON t_prompt.id = bc.chatwoot_team_id
LEFT JOIN chatwoot.teams t_fallback
    ON t_fallback.id = bc.fallback_team_id
ORDER BY bc.enabled DESC, bc.updated_at DESC NULLS LAST, bc.id DESC;
